10 // return false if pid is not supported, true if it is.
11 // mode is 0 for get_pid() and 1 for menu config to allow pid > 0xF0
12 bool is_pid_supported(unsigned char pid, unsigned char mode)
13 {
14  if(pid==0)
15  return true;
16  else
17  if(pid<=0x20)
18  {
19  if(1L<<(uint8_t)(0x20-pid) & pid01to20_support)
20  return true;
21  }
22  else
23  if(pid<=0x40)
24  {
25  if(1L<<(uint8_t)(0x40-pid) & pid21to40_support)
26  return true;
27  }
28  else
29  if(pid<=0x60)
30  {
31  if(1L<<(uint8_t)(0x60-pid) & pid41to60_support)
32  return true;
33  }
34  else
35  if( mode && pid>=FIRST_FAKE_PID)
36  return true;
37 
38  return false;
39 }
