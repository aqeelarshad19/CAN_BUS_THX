117  void RESET001_AssertReset(RESET001_ResetnType Resetn)

125 
126  SCU_RESET_TypeDef* RCUCtrlReg = SCU_RESET;
127 
128  /* Extract the MSB to identify the peripheral in which the module is residing.
129  * Shift the MSB to LSB position and store it in a local variable Temp */
130  PeripheralInfo = (((uint32_t)Resetn & RESET001_CLEAR_BITMASK ) >>
131  RESET001_SHIFT_RIGHT_MASK);
132 
133  /* Multiply the peripheral info stored in Temp variable with the corresponding

135  * Add the offset address with base PRSET peripheral address to get the actual
136  * RCU PRSET register address */
137  RCUControlReg = (uint32_t*)((uint32_t)(&RCUCtrlReg->PRSET0) +
138  ( RESET001_PERIPHERAL_ADDR_SHIFT * PeripheralInfo));
139 
140  /* Load the PRSET register with the reset value after ignoring the Most
141  * Significant Nibble. */
142  *RCUControlReg = ((uint32_t)Resetn & RESET001_BITMASK );
