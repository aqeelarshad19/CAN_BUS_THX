129 {
130  static uint8_t isInitialized = 0U;
131  if(isInitialized == 0U)
132  {
133  /* Reset MultiCAN module */
134  RESET001_DeassertReset(PER1_MCAN0);
135  /* Enable CAN Module */
136  CAN->CLC &= (uint32_t)~CAN_CLC_DISR_Msk;
137  /* Select the Divider Mode */
